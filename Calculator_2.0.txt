    // ---- Time helpers ----
    function hmToMinutes(hm) {
      const [h, m] = hm.split(':').map(Number);
      return h * 60 + m;
    }
    function minutesOfDay(d) {
      return d.getHours() * 60 + d.getMinutes();
    }
    function sameDate(a, b) {
      return a.getFullYear() === b.getFullYear() && a.getMonth() === b.getMonth() && a.getDate() === b.getDate();
    }
    function atTimeOnDate(dateObj, hhmm) {
      const [h, m] = hhmm.split(':').map(Number);
      const d = new Date(dateObj);
      d.setHours(h, m, 0, 0);
      return d;
    }
    function addDays(dateObj, days) {
      const d = new Date(dateObj);
      d.setDate(d.getDate() + days);
      return d;
    }
    function ceilDiv(a, b) {
      return Math.ceil(a / b);
    }

    // ---- Pricing engine (full rules) ----
    function priceStay(start, end, hasConstantCompany, discountCodeRaw) {
      const code = (discountCodeRaw || '').trim().toUpperCase();
      const discountCodes = ['KASIA5', 'GRANDFATHERED'];

      const useDiscount = discountCodes.includes(code);

      // Base services
      const base = {
        fullDayOvernight: 35,
        extraDaycare: 25,
        daycareOnly: 30,
        overnightOnly: 25
      };
      const discounted = {
        fullDayOvernight: 30,
        extraDaycare: 20,
        daycareOnly: 25,
        overnightOnly: 20
      };

      const rates = useDiscount ? discounted : base;

      // Windows (minutes)
      const WIN = {
        fullDayDropFrom: hmToMinutes('08:00'),
        fullDayDropTo: hmToMinutes('12:00'),

        daycareFrom: hmToMinutes('08:00'),
        daycareTo: hmToMinutes('22:00'),
        daycareMaxMinutes: 12 * 60,

        overnightDropFrom: hmToMinutes('19:00'),
        overnightDropTo: hmToMinutes('22:00'),
        overnightPickupTo: hmToMinutes('10:00'),

        extraDaycarePickupTo: hmToMinutes('19:00'),

        latePickupFrom: hmToMinutes('22:00'),
        latePickupRatePerStartedHour: 3,

        constantCompanyPerStarted24h: 5
      };

      // Late pickup fee (applied based on final pickup time after 22:00)
      function latePickupFee(finalEnd) {
        const cutoff = atTimeOnDate(finalEnd, '22:00');
        if (finalEnd <= cutoff) return { fee: 0, startedHours: 0 };
        const ms = finalEnd - cutoff;
        const startedHours = Math.ceil(ms / (1000 * 60 * 60));
        return { fee: startedHours * WIN.latePickupRatePerStartedHour, startedHours };
      }

      // Constant company fee: €5 per started 24h block
      function constantCompanyFee(totalMinutes) {
        const startedBlocks = ceilDiv(totalMinutes, 24 * 60);
        return { fee: startedBlocks * WIN.constantCompanyPerStarted24h, startedBlocks };
      }

      // Coverage check: are start/end within any allowed “shape”?
      // We still compute a best estimate if not, but show a notice.
      function isClearlyWithinWindows(start, end) {
        const sMin = minutesOfDay(start);
        const eMin = minutesOfDay(end);

        // Daycare-only shape (same day, end <= 22, duration <= 12h, start >= 8)
        const totalMin = (end - start) / (1000 * 60);
        if (sameDate(start, end) &&
            sMin >= WIN.daycareFrom &&
            eMin <= WIN.daycareTo &&
            totalMin <= WIN.daycareMaxMinutes) return true;

        // Overnight-only shape (start 19-22, end next day <= 10)
        if (!sameDate(start, end)) {
          const nextDay = addDays(start, 1);
          if (sameDate(end, nextDay) &&
              sMin >= WIN.overnightDropFrom &&
              sMin <= WIN.overnightDropTo &&
              eMin <= WIN.overnightPickupTo) return true;
        }

        // Full-day blocks require drop-off 08-12 (we can still estimate even if not)
        if (sMin >= WIN.fullDayDropFrom && sMin <= WIN.fullDayDropTo) return true;

        return false;
      }

      // Dynamic programming:
      // State = current time + a flag whether we "just had an overnight" so extra daycare can apply.
      // We explore legal next segments and choose the cheapest way to reach >= end.
      const memo = new Map();

      function key(t, hadOvernight) {
        return `${t.getTime()}|${hadOvernight ? 1 : 0}`;
      }

      function solve(t, hadOvernight) {
        const k = key(t, hadOvernight);
        if (memo.has(k)) return memo.get(k);

        if (t >= end) {
          const res = { cost: 0, items: [] };
          memo.set(k, res);
          return res;
        }

        const candidates = [];

        const tMin = minutesOfDay(t);

        // 1) Daycare-only segment (as a remainder block): can cover up to 12h but not past 22:00 same day
        // We use it as a segment if it gets us closer (fixed fee).
        if (tMin >= WIN.daycareFrom && tMin <= WIN.daycareTo) {
          const maxEndByDuration = new Date(t.getTime() + WIN.daycareMaxMinutes * 60 * 1000);
          const endOfDaycareWindow = atTimeOnDate(t, '22:00');
          const segEnd = new Date(Math.min(maxEndByDuration.getTime(), endOfDaycareWindow.getTime()));

          // Only add if it actually advances time
          if (segEnd > t) {
            const next = solve(segEnd, false);
            candidates.push({
              cost: rates.daycareOnly + next.cost,
              items: [{ nameKey: 'daycareOnly', label: 'Daycare only', price: rates.daycareOnly, from: t, to: segEnd }].concat(next.items)
            });
          }
        }

        // 2) Overnight-only segment (19:00–10:00)
        if (tMin >= WIN.overnightDropFrom && tMin <= WIN.overnightDropTo) {
          const overnightEnd = atTimeOnDate(addDays(t, 1), '10:00');
          if (overnightEnd > t) {
            const next = solve(overnightEnd, true);
            candidates.push({
              cost: rates.overnightOnly + next.cost,
              items: [{ nameKey: 'overnightOnly', label: 'Overnight only', price: rates.overnightOnly, from: t, to: overnightEnd }].concat(next.items)
            });
          }
        }

        // 3) Full day + overnight (24h) — only "official" if drop-off is 08:00–12:00
        if (tMin >= WIN.fullDayDropFrom && tMin <= WIN.fullDayDropTo) {
          const fullEnd = new Date(t.getTime() + 24 * 60 * 60 * 1000);
          const next = solve(fullEnd, true); // counts as having had an overnight
          candidates.push({
            cost: rates.fullDayOvernight + next.cost,
            items: [{ nameKey: 'fullDayOvernight', label: 'Full day + overnight', price: rates.fullDayOvernight, from: t, to: fullEnd }].concat(next.items)
          });
        }

        // 4) Extra daycare after an overnight — until 19:00 same day
        if (hadOvernight) {
          const extraEnd = atTimeOnDate(t, '19:00');
          if (extraEnd > t) {
            const next = solve(extraEnd, false);
            candidates.push({
              cost: rates.extraDaycare + next.cost,
              items: [{ nameKey: 'extraDaycare', label: 'Extra daycare', price: rates.extraDaycare, from: t, to: extraEnd }].concat(next.items)
            });
          }
        }

        // 5) Fallback: if we have NO candidates (times outside windows), advance by 24h blocks estimator
        // This prevents “no solution” for real-life dropoffs (e.g. 18:30).
        if (candidates.length === 0) {
          const fullEnd = new Date(t.getTime() + 24 * 60 * 60 * 1000);
          const next = solve(fullEnd, true);
          candidates.push({
            cost: rates.fullDayOvernight + next.cost,
            items: [{ nameKey: 'fullDayOvernight', label: 'Full day + overnight', price: rates.fullDayOvernight, from: t, to: fullEnd, isFallback: true }].concat(next.items)
          });
        }

        // Choose cheapest candidate
        candidates.sort((a, b) => a.cost - b.cost);
        const best = candidates[0];
        memo.set(k, best);
        return best;
      }

      const basePlan = solve(start, false);

      // Now trim plan items so it doesn’t list segments entirely after end
      const items = [];
      for (const it of basePlan.items) {
        if (it.from >= end) break;
        // Keep item if it overlaps the actual stay window
        if (it.to > start) items.push(it);
      }

      // Total minutes for constant company
      const totalMinutes = (end - start) / (1000 * 60);

      // Add constant company + late pickup
      let total = basePlan.cost;
      const breakdown = [];

      // Add base services breakdown
      for (const it of items) {
        let name = it.label;
        if (it.isFallback) name += ' (estimate)';
        breakdown.push({ type: 'service', name, price: it.price });
      }

      // Constant company
      let cc = { fee: 0, startedBlocks: 0 };
      if (hasConstantCompany) {
        cc = constantCompanyFee(totalMinutes);
        total += cc.fee;
      }

      // Late pickup
      const late = latePickupFee(end);
      total += late.fee;

      return {
        total,
        breakdown,
        useDiscount,
        discountCode: useDiscount ? code : '',
        lateStartedHours: late.startedHours,
        lateFee: late.fee,
        constantCompanyStartedBlocks: cc.startedBlocks,
        constantCompanyFee: cc.fee,
        clearlyWithinWindows: isClearlyWithinWindows(start, end)
      };
    }

    function calculate() {
      const t = i18n[currentLang];
      const errorEl = document.getElementById('error');
      const noticeEl = document.getElementById('notice');
      const resultEl = document.getElementById('result');

      errorEl.style.display = 'none';
      noticeEl.style.display = 'none';
      resultEl.style.display = 'none';
      errorEl.textContent = '';
      noticeEl.textContent = '';
      resultEl.textContent = '';

      const dropDate = document.getElementById('dropDate').value;
      const dropTime = document.getElementById('dropTime').value;
      const pickDate = document.getElementById('pickDate').value;
      const pickTime = document.getElementById('pickTime').value;
      const constantCompany = document.getElementById('constantCompany').checked;
      const discountCode = document.getElementById('discountCode').value;

      if (!dropDate || !dropTime || !pickDate || !pickTime) {
        errorEl.textContent = t.errMissing;
        errorEl.style.display = 'block';
        return;
      }

      const start = new Date(`${dropDate}T${dropTime}:00`);
      const end = new Date(`${pickDate}T${pickTime}:00`);

      if (isNaN(start.getTime()) || isNaN(end.getTime())) {
        errorEl.textContent = t.errBad;
        errorEl.style.display = 'block';
        return;
      }
      if (end <= start) {
        errorEl.textContent = t.errOrder;
        errorEl.style.display = 'block';
        return;
      }

      const res = priceStay(start, end, constantCompany, discountCode);

      if (!res.clearlyWithinWindows) {
        noticeEl.textContent = t.noticeOutOfWindow;
        noticeEl.style.display = 'block';
      }

      const hours = (end - start) / (1000 * 60 * 60);
      const started24h = Math.ceil(((end - start) / (1000 * 60)) / (24 * 60));

      const lines = [];
      lines.push(t.lineDuration(hours));
      if (constantCompany) lines.push(t.lineBlocks(started24h));
      lines.push('');

      // Summarize service lines (group by name)
      const grouped = new Map();
      for (const b of res.breakdown) {
        const key = b.name;
        grouped.set(key, (grouped.get(key) || 0) + b.price);
      }
      for (const [name, sum] of grouped.entries()) {
        lines.push(t.lineItem(name, sum));
      }

      if (constantCompany && res.constantCompanyFee > 0) {
        lines.push(t.lineItem(currentLang === 'en' ? 'Constant company' : 'Constant company', res.constantCompanyFee));
      }

      if (res.lateFee > 0) {
        lines.push(t.lineLate(res.lateStartedHours, res.lateFee));
      }

      if (res.useDiscount) {
        lines.push('');
        lines.push(t.lineDiscount(res.discountCode));
      }

      lines.push('');
      lines.push(t.total(res.total));

      resultEl.textContent = lines.join('\n');
      resultEl.style.display = 'block';
    }

    setLanguage('en');
  </script>
